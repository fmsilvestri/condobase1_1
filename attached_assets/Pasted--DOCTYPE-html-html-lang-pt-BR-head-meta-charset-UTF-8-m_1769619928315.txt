    <script>
        // Vari√°veis globais
        let templateSelecionado = null;
        let membrosSelecionados = [];
        let templates = [];

        // Inicializa√ß√£o
        document.addEventListener('DOMContentLoaded', () => {

        // Carregar templates na UI
        async function carregarTemplatesUI() {
            templates = await carregarTemplates();
            exibirTemplates(templates);
            popularSelectTemplates(templates);
        }

        // Exibir templates
        function exibirTemplates(templates) {
            const container = document.getElementById('templatesContainer');
            container.innerHTML = '';

            templates.forEach(template => {
                const card = criarCardTemplate(template);
                container.appendChild(card);
            });
        }

        // Criar card de template
        function criarCardTemplate(template) {
            const card = document.createElement('div');
            card.className = 'template-card';
            card.onclick = () => selecionarTemplate(template.id, card);

            card.innerHTML = `
                <div class="template-header">
                    <div class="template-titulo">${template.nome}</div>
                    <span class="template-badge">${template.funcao}</span>
                </div>
                <div class="template-descricao">${template.descricao || ''}</div>
                <div class="template-info">
                    <span class="template-info-item">
                        üìÖ ${template.periodicidade}
                    </span>
                    <span class="template-info-item">
                        ‚è∞ ${template.turno_padrao}
                    </span>
                    <span class="template-info-item">
                        üìä Usado ${template.uso_count}x
                    </span>
                </div>
            `;

            return card;
        }

        // Selecionar template
        function selecionarTemplate(templateId, card) {
            document.querySelectorAll('.template-card').forEach(c => {
                c.classList.remove('selected');
            });
            card.classList.add('selected');
            templateSelecionado = templateId;
        }

        // Popular select de templates
        function popularSelectTemplates(templates) {
            const select = document.getElementById('templateEnvio');
            select.innerHTML = '<option value="">Selecione...</option>';
            
            templates.forEach(template => {
                const option = document.createElement('option');
                option.value = template.id;
                option.textContent = `${template.nome} (${template.funcao})`;
                select.appendChild(option);
            });
        }

        // Carregar membros
        async function carregarMembros() {
            const { data, error } = await supabase
                .from('membros_equipe')
                .select('*')
                .eq('status', 'ativo')
                .order('nome');

            if (!error && data) {
                exibirMembros(data);
            }
        }

        // Exibir membros
        function exibirMembros(membros) {
            const container = document.getElementById('membrosContainer');
            container.innerHTML = '';

            membros.forEach(membro => {
                const item = document.createElement('div');
                item.className = 'membro-item';
                item.onclick = () => toggleMembro(membro.id, item);

                item.innerHTML = `
                    <input type="checkbox" class="membro-checkbox" data-membro-id="${membro.id}">
                    <div class="membro-info">
                        <div class="membro-nome">${membro.nome}</div>
                        <div class="membro-funcao">${membro.funcao}</div>
                        ${membro.whatsapp ? `<div class="membro-whatsapp">üì± ${membro.whatsapp}</div>` : ''}
                    </div>
                `;

                container.appendChild(item);
            });
        }

        // Toggle sele√ß√£o de membro
        function toggleMembro(membroId, item) {
            const checkbox = item.querySelector('.membro-checkbox');
            checkbox.checked = !checkbox.checked;
            
            if (checkbox.checked) {
                item.classList.add('selected');
                membrosSelecionados.push(membroId);
            } else {
                item.classList.remove('selected');
                membrosSelecionados = membrosSelecionados.filter(id => id !== membroId);
            }
        }

        // Visualizar PDF
        function visualizarPDF() {
            alert('Preview do PDF ser√° implementado aqui');
            document.getElementById('modalPDF').classList.add('active');
        }

        // Fechar modal
        function fecharModal() {
            document.getElementById('modalPDF').classList.remove('active');
        }

        // Mudar tab
        function mudarTab(tab) {
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            
            event.target.classList.add('active');
            document.getElementById(`tab-${tab}`).classList.add('active');
        }

        // Set data padr√£o
        function setDataPadrao() {
            const hoje = new Date().toISOString().split('T')[0];
            document.getElementById('dataExecucaoEnvio').value = hoje;
        }

        // Filtrar templates
        function filtrarTemplates() {
            const funcao = document.getElementById('filtroFuncao').value;
            const periodicidade = document.getElementById('filtroPeriodicidade').value;
            
            let filtrados = templates;
            
            if (funcao) {
                filtrados = filtrados.filter(t => t.funcao === funcao);
            }
            
            if (periodicidade) {
                filtrados = filtrados.filter(t => t.periodicidade === periodicidade);
            }
            
            exibirTemplates(filtrados);
        }

        // Submit do formul√°rio
        document.getElementById('formEnvioLote').addEventListener('submit', async (e) => {
            e.preventDefault();
            
            if (membrosSelecionados.length === 0) {
                alert('Selecione pelo menos um funcion√°rio');
                return;
            }

            const dados = {
                templateId: document.getElementById('templateEnvio').value,
                membrosIds: membrosSelecionados,
                dataExecucao: document.getElementById('dataExecucaoEnvio').value,
                turno: document.getElementById('turnoEnvio').value,
                observacoes: document.getElementById('observacoesEnvio').value
            };

            const resultado = await enviarListasEmLote(dados);
            
            if (resultado.sucesso) {
                alert('Listas enviadas com sucesso!');
                location.reload();
            } else {
                alert('Erro: ' + resultado.erro);
            }
        });
