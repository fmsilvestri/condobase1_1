    <script>
        // Configurar Supabase (substitua com suas credenciais)
        const SUPABASE_URL = 'SUA_URL_AQUI';
        const SUPABASE_KEY = 'SUA_CHAVE_AQUI';
        const supabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);

        // Fun√ß√µes de formata√ß√£o
        function formatarCPF(input) {
            let valor = input.value.replace(/\D/g, '');
            if (valor.length <= 11) {
                valor = valor.replace(/(\d{3})(\d)/, '$1.$2');
                valor = valor.replace(/(\d{3})(\d)/, '$1.$2');
                valor = valor.replace(/(\d{3})(\d{1,2})$/, '$1-$2');
            }
            input.value = valor;
        }

        function formatarTelefone(input) {
            let valor = input.value.replace(/\D/g, '');
            if (valor.length <= 11) {
                if (valor.length === 11) {
                    valor = valor.replace(/(\d{2})(\d{5})(\d{4})/, '($1) $2-$3');
                } else if (valor.length === 10) {
                    valor = valor.replace(/(\d{2})(\d{4})(\d{4})/, '($1) $2-$3');
                }
            }
            input.value = valor;
        }

        // Aplicar m√°scaras
        document.getElementById('cpf').addEventListener('input', function() {
            formatarCPF(this);
        });

        document.getElementById('telefone').addEventListener('input', function() {
            formatarTelefone(this);
        });

        document.getElementById('whatsapp').addEventListener('input', function() {
            formatarTelefone(this);
        });

        // Fun√ß√£o para mostrar alertas
        function showAlert(type, message) {
            const alertSuccess = document.getElementById('alertSuccess');
            const alertError = document.getElementById('alertError');
            
            if (type === 'success') {
                alertSuccess.classList.add('show');
                alertError.classList.remove('show');
                setTimeout(() => alertSuccess.classList.remove('show'), 5000);
            } else {
                alertError.classList.add('show');
                alertSuccess.classList.remove('show');
                document.getElementById('errorMessage').textContent = message;
                setTimeout(() => alertError.classList.remove('show'), 5000);
            }
        }

        // Submit do formul√°rio
        document.getElementById('formNovoMembro').addEventListener('submit', async (e) => {
            e.preventDefault();

            const btnSubmit = document.getElementById('btnSubmit');
            btnSubmit.textContent = 'Cadastrando...';
            btnSubmit.disabled = true;

            try {
                const formData = {
                    nome: document.getElementById('nome').value.toUpperCase(),
                    cpf: document.getElementById('cpf').value,
                    funcao: document.getElementById('funcao').value,
                    departamento: document.getElementById('departamento').value || null,
                    status: document.getElementById('status').value,
                    telefone: document.getElementById('telefone').value || null,
                    whatsapp: document.getElementById('whatsapp').value || null,
                    email: document.getElementById('email').value || null,
                    horario_trabalho: document.getElementById('horario_trabalho').value || null,
                    data_contratacao: document.getElementById('data_contratacao').value || null,
                    observacoes: document.getElementById('observacoes').value || null
                };

                const { data, error } = await supabase
                    .from('membros_equipe')
                    .insert([formData])
                    .select();

                if (error) throw error;

                showAlert('success');
                e.target.reset();

            } catch (error) {
                console.error('Erro:', error);
                let errorMsg = error.message;
                
                if (error.code === '23505') {
                    errorMsg = 'Este CPF j√° est√° cadastrado no sistema';
                }
                
                showAlert('error', errorMsg);
            } finally {
                btnSubmit.textContent = '‚ûï Cadastrar Membro';
                btnSubmit.disabled = false;
            }
        });
    </script>
