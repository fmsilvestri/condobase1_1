    <script>
        // Vari√°veis globais
        let atividadesSelecionadas = [];
        let membroAtual = null;

        // Inicializa√ß√£o
        document.addEventListener('DOMContentLoaded', () => {
            carregarMembros();
            setDataPadrao();
        });

        // Carregar membros da equipe
        async function carregarMembros() {
            const { data, error } = await supabase
                .from('membros_equipe')
                .select('*')
                .eq('status', 'ativo')
                .order('nome');

            if (!error && data) {
                const select = document.getElementById('membroSelect');
                data.forEach(membro => {
                    const option = document.createElement('option');
                    option.value = membro.id;
                    option.textContent = `${membro.nome} - ${membro.funcao}`;
                    option.dataset.membro = JSON.stringify(membro);
                    select.appendChild(option);
                });
            }
        }

        // Carregar atividades quando selecionar membro
        async function carregarAtividades() {
            const select = document.getElementById('membroSelect');
            const option = select.options[select.selectedIndex];
            
            if (!option.value) return;
            
            membroAtual = JSON.parse(option.dataset.membro);
            atividadesSelecionadas = [];

            const atividades = await carregarAtividadesPorFuncao(membroAtual.funcao);
            
            const container = document.getElementById('atividadesContainer');
            container.innerHTML = '';

            if (atividades.length === 0) {
                container.innerHTML = '<p>Nenhuma atividade cadastrada para esta fun√ß√£o</p>';
                return;
            }

            atividades.forEach(ativ => {
                const card = criarCardAtividade(ativ);
                container.appendChild(card);
            });
        }

        // Criar card de atividade
        function criarCardAtividade(atividade) {
            const card = document.createElement('div');
            card.className = 'atividade-card';
            card.onclick = () => toggleAtividade(atividade, card);

            let html = `
                <div class="atividade-header">
                    <div>
                        <div class="atividade-titulo">${atividade.titulo}</div>
                        <div style="font-size: 12px; color: #999; margin-top: 5px;">
                            ${atividade.periodicidade} ‚Ä¢ ${atividade.duracao_estimada}min
                        </div>
                    </div>
                    <span class="categoria-badge" style="background: ${atividade.categoria.cor}">
                        ${atividade.categoria.icone}
                    </span>
                </div>
                <p style="font-size: 13px; color: #666; margin-top: 10px;">
                    ${atividade.descricao || ''}
                </p>
            `;

            if (atividade.checklist && atividade.checklist.items) {
                html += '<div class="checklist">';
                atividade.checklist.items.slice(0, 3).forEach(item => {
                    html += `<div class="checklist-item">‚Ä¢ ${item}</div>`;
                });
                if (atividade.checklist.items.length > 3) {
                    html += `<div class="checklist-item">... e mais ${atividade.checklist.items.length - 3}</div>`;
                }
                html += '</div>';
            }

            if (atividade.equipamentos_necessarios && atividade.equipamentos_necessarios.length > 0) {
                html += `<div class="equipamentos">üß∞ ${atividade.equipamentos_necessarios.join(', ')}</div>`;
            }

            card.innerHTML = html;
            return card;
        }

        // Toggle sele√ß√£o de atividade
        function toggleAtividade(atividade, card) {
            const index = atividadesSelecionadas.findIndex(a => a.id === atividade.id);
            
            if (index > -1) {
                atividadesSelecionadas.splice(index, 1);
                card.classList.remove('selected');
            } else {
                atividadesSelecionadas.push(atividade);
                card.classList.add('selected');
            }
        }

        // Visualizar mensagem
        function visualizarMensagem() {
            if (!membroAtual) {
                alert('Selecione um membro primeiro');
                return;
            }

            if (atividadesSelecionadas.length === 0) {
                alert('Selecione pelo menos uma atividade');
                return;
            }

            const lista = {
                titulo: document.getElementById('tituloLista').value,
                data_execucao: document.getElementById('dataExecucao').value,
                turno: document.getElementById('turno').value,
                prioridade: document.getElementById('prioridade').value,
                observacoes: document.getElementById('observacoes').value
            };

            const mensagem = gerarMensagemWhatsApp(lista, membroAtual, atividadesSelecionadas);
            
            document.getElementById('previewContent').textContent = mensagem;
            document.getElementById('modalPreview').classList.add('active');
        }

        // Mudar tab
        function mudarTab(tab) {
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            
            event.target.classList.add('active');
            document.getElementById(`tab-${tab}`).classList.add('active');
        }

        // Fechar modal
        function fecharModal() {
            document.getElementById('modalPreview').classList.remove('active');
        }

        // Set data padr√£o (hoje)
        function setDataPadrao() {
            const hoje = new Date().toISOString().split('T')[0];
            document.getElementById('dataExecucao').value = hoje;
        }

        // Submit do formul√°rio
        document.getElementById('formNovaLista').addEventListener('submit', async (e) => {
            e.preventDefault();
            
            if (atividadesSelecionadas.length === 0) {
                alert('Selecione pelo menos uma atividade');
                return;
            }

            const dados = {
                membroId: document.getElementById('membroSelect').value,
                titulo: document.getElementById('tituloLista').value,
                dataExecucao: document.getElementById('dataExecucao').value,
                turno: document.getElementById('turno').value,
                prioridade: document.getElementById('prioridade').value,
                observacoes: document.getElementById('observacoes').value,
                atividades: atividadesSelecionadas
            };

            const resultado = await criarListaAtividades(dados);
            
            if (resultado.sucesso) {
                await enviarParaWhatsApp(resultado.lista.id, dados.membroId);
                alert('Lista criada e enviada com sucesso!');
                location.reload();
            } else {
                alert('Erro: ' + resultado.erro);
            }
        });
